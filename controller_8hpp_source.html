237  float step(float reference[3], float estimate[3]) {
238  float dt = timer.delta();
239  float output = 0.0;
240 
241  pid1.K[0] = gains[0];
242  pid1.K[1] = gains[1];
243  pid1.K[2] = gains[2];
244  pid1.K[3] = gains[3] * sin(reference[0]);
245  pid2.K[0] = gains[4];
246  pid2.K[1] = gains[5];

250  pid1.measurement = estimate[0];
251 
252  pid2.setpoint = reference[1];
253  pid2.measurement = estimate[1];
254 
255  output += pid1.filter(dt, true, true); // position wraps
256  output += pid2.filter(dt, true, false); // no wrap for velocity
257  output = constrain(output, -1.0, 1.0);
258  return output;
259  }
